let valor = prompt("Digite o valor a ser sacado")

let resto = Number(valor)

while(resto >= 100){
	document.write("100")
	resto = resto - 100
}
while(resto >= 50){
	document.write("50")
	resto = resto - 50
}
while(resto >= 10){
	document.write("10")
	resto = resto - 10
}
while(resto >= 5){
	document.write("5")
	resto = resto - 5
}
while(resto >= 1){
	document.write("1")
	resto = resto - 1
}
